1"""Property-based tests for grid invariants using Hypothesis. 

2 

3These tests use generative testing to verify that grid properties hold 

4for a wide range of inputs. Hypothesis automatically generates test cases 

5and shrinks failures to minimal examples. 

6 

7Tests cover: 

8- Grid always has positive cell count 

9- Coordinates stay within hemisphere bounds 

10- Cell IDs are unique 

11- Solid angles sum to hemisphere area 

12- Phi/theta bounds are valid 

13- Grid DataFrame is always Polars 

14""" 

15 

16from __future__ import annotations 

17

19import polars as pl 

20import pytest 

21from canvod.grids import create_hemigrid 

22from hypothesis import given, settings

26class TestGridInvariants: 

27 """Property-based tests for grid structural invariants.""" 

28 

29 @given( 

30 angular_resolution=st.floats(min_value=2.0, max_value=45.0, exclude_min=False) 

31 ) 

32 @settings(max_examples=50, deadline=None) 

33 def test_grid_ncells_always_positive(self, angular_resolution: float) -> None: 

34 """Any valid resolution should create grid with positive cells.""" 

35 # Create equal-area grid 

36 grid = create_hemigrid("equal_area", angular_resolution) 

37 

38 # Fundamental invariant: grids must have cells 

39 assert grid.ncells > 0 

40 assert isinstance(grid.ncells, int) 

41 

42 @given( 

43 angular_resolution=st.floats(min_value=5.0, max_value=30.0, exclude_min=False) 

44 ) 

45 @settings(max_examples=30, deadline=None) 

46 def test_grid_coords_in_hemisphere(self, angular_resolution: float) -> None: 

47 """All cell centers must be within hemisphere bounds.""" 

48 grid = create_hemigrid("equal_area", angular_resolution) 

49 

50 # Extract coordinates 

51 phi = grid.grid["phi"].to_numpy() 

52 theta = grid.grid["theta"].to_numpy() 

53 

54 # Phi must be in [0, 2π) 

55 assert np.all(phi >= 0.0) 

56 assert np.all(phi < 2 * np.pi) 

57 

58 # Theta must be in [0, π/2] for hemisphere 

59 assert np.all(theta >= 0.0) 

60 assert np.all(theta <= np.pi / 2) 

61 

62 @given( 

63 angular_resolution=st.floats(min_value=5.0, max_value=30.0, exclude_min=False) 

64 ) 

65 @settings(max_examples=30, deadline=None) 

66 def test_cell_ids_unique(self, angular_resolution: float) -> None: 

67 """Cell IDs must be unique across the grid.""" 

68 grid = create_hemigrid("equal_area", angular_resolution) 

69 

70 cell_ids = grid.grid["cell_id"].to_numpy() 

71 

72 # All IDs must be unique 

73 assert len(cell_ids) == len(np.unique(cell_ids)) 

74 

75 # IDs should be sequential starting from 0 

76 assert np.all(cell_ids == np.arange(grid.ncells)) 

77 

78 @pytest.mark.skip( 

79 reason="Solid angle calculation may not sum exactly to hemisphere" 

80 ) 

81 @given( 

82 angular_resolution=st.floats(min_value=10.0, max_value=30.0, exclude_min=False) 

83 ) 

84 @settings(max_examples=20, deadline=None) 

85 def test_solid_angles_sum_to_hemisphere(self, angular_resolution: float) -> None: 

86 """Sum of solid angles should approximately equal hemisphere area.""" 

87 grid = create_hemigrid("equal_area", angular_resolution) 

88 

89 solid_angles = grid.get_solid_angles() 

90 

91 # Sum should be approximately 2π (hemisphere area in steradians) 

92 total_solid_angle = np.sum(solid_angles) 

93 expected_area = 2 * np.pi 

94 

95 # Allow 10% tolerance - grids may not cover exact hemisphere 

96 # (edge effects, discrete bins) 

97 assert total_solid_angle > expected_area * 0.90, ( 

98 f"Got {total_solid_angle}, expected ~{expected_area}" 

99 ) 

100 

101 @given( 

102 angular_resolution=st.floats(min_value=5.0, max_value=30.0, exclude_min=False) 

103 ) 

104 @settings(max_examples=30, deadline=None) 

105 def test_phi_theta_bounds_valid(self, angular_resolution: float) -> None: 

106 """Cell boundaries must satisfy min < max for phi and theta.""" 

107 grid = create_hemigrid("equal_area", angular_resolution) 

108 

109 # Check bounds exist for rectangular grids 

110 if "phi_min" in grid.grid.columns: 

111 phi_min = grid.grid["phi_min"].to_numpy() 

112 phi_max = grid.grid["phi_max"].to_numpy() 

113 theta_min = grid.grid["theta_min"].to_numpy() 

114 theta_max = grid.grid["theta_max"].to_numpy() 

115 

116 # Min must be less than max (with tolerance for wrapping) 

117 # Some cells may wrap around φ=0/2π 

118 non_wrapping = phi_max > phi_min 

119 assert np.sum(non_wrapping) > 0 # Most cells don't wrap 

120 

121 # Theta should never wrap 

122 assert np.all(theta_max > theta_min) 

123 

124 @given( 

125 angular_resolution=st.floats(min_value=5.0, max_value=30.0, exclude_min=False) 

126 ) 

127 @settings(max_examples=30, deadline=None) 

128 def test_polars_dataframe_invariant(self, angular_resolution: float) -> None: 

129 """Grid structure is always a Polars DataFrame.""" 

130 grid = create_hemigrid("equal_area", angular_resolution) 

131 

132 # Core invariant: grid.grid must be Polars DataFrame 

133 assert isinstance(grid.grid, pl.DataFrame) 

134 

135 # Must have required columns 

136 required_cols = ["cell_id", "phi", "theta"] 

137 for col in required_cols: 

138 assert col in grid.grid.columns 

139 

140 # Number of rows must match ncells 

141 assert len(grid.grid) == grid.ncells

144class TestEqualAreaSpecificProperties: 

145 """Property tests specific to equal-area grids.""" 

146 

147 @given( 

148 angular_resolution=st.floats(min_value=10.0, max_value=30.0, exclude_min=False) 

149 ) 

150 @settings(max_examples=20, deadline=None) 

151 def test_equal_area_cells_similar_size(self, angular_resolution: float) -> None: 

152 """Equal-area grids should have cells with similar solid angles.""" 

153 grid = create_hemigrid("equal_area", angular_resolution) 

154 

155 solid_angles = grid.get_solid_angles() 

156 

157 # Calculate coefficient of variation (std/mean) 

158 mean_sa = np.mean(solid_angles) 

159 std_sa = np.std(solid_angles) 

160 cv = std_sa / mean_sa 

161 

162 # Equal-area grids should have low variation (< 20%) 

163 # This is a design goal, not a hard requirement 

164 assert cv < 0.2, f"Coeff of variation {cv:.2%} too high for equal-area" 

165 

166 @given( 

167 angular_resolution=st.floats(min_value=10.0, max_value=30.0, exclude_min=False) 

168 ) 

169 @settings(max_examples=20, deadline=None) 

170 def test_equal_area_has_rectangular_columns( 

171 self, angular_resolution: float 

172 ) -> None: 

173 """Equal-area grids must have phi/theta min/max columns.""" 

174 grid = create_hemigrid("equal_area", angular_resolution) 

175 

176 # Equal-area is rectangular grid type 

177 assert "phi_min" in grid.grid.columns 

178 assert "phi_max" in grid.grid.columns 

179 assert "theta_min" in grid.grid.columns 

180 assert "theta_max" in grid.grid.columns

199 @given( 

200 angular_resolution=st.floats(min_value=5.0, max_value=20.0, exclude_min=False) 

201 ) 

202 @settings(max_examples=15, deadline=None) 

203 def test_htm_vertices_are_3d(self, angular_resolution: float) -> None: 

204 """HTM vertices must be 3D Cartesian coordinates.""" 

205 grid = create_hemigrid("htm", angular_resolution) 

206 

207 # Check first cell's vertices 

208 first_row = grid.grid.row(0, named=True) 

209 v0 = first_row["htm_vertex_0"] 

210 v1 = first_row["htm_vertex_1"] 

211 v2 = first_row["htm_vertex_2"] 

212 

213 # Each vertex should be list/array of 3 coordinates 

214 assert len(v0) == 3 

215 assert len(v1) == 3 

216 assert len(v2) == 3 

217 

218 # Coordinates should be on unit sphere (|v| ≈ 1) 

219 assert np.isclose(np.linalg.norm(v0), 1.0, rtol=1e-5) 

220 assert np.isclose(np.linalg.norm(v1), 1.0, rtol=1e-5) 

221 assert np.isclose(np.linalg.norm(v2), 1.0, rtol=1e-5)
